📎 Google Apps Script — paste this once

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  
  // Add header row if sheet is empty
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "Timestamp","Name","Date","Wake Time","Sleep Hrs","Weight(kg)",
      "Stress","Energy","Morning Notes","Water(L)","Caffeine","Other Liquids",
      "Meals Log","Workout","Duration","Workout Time","Intensity","Workout Notes",
      "Digestion","Symptoms","Symptom Notes","Hunger","Bedtime",
      "Cravings","Day Rating","Final Notes"
    ]);
  }
  
  sheet.appendRow([
    new Date().toLocaleString(), data.name, data.date,
    data.wakeTime, data.sleepHours, data.weight,
    data.stressLevel, data.energyLevel, data.morningNotes,
    data.waterIntake, data.caffeineCount, data.otherLiquids,
    data.meals, data.workoutType, data.workoutDuration,
    data.workoutTime, data.workoutIntensity, data.workoutNotes,
    data.digestion, data.symptoms, data.symptomNotes,
    data.hungerLevel, data.bedTime, data.cravings,
    data.dayRating, data.finalNotes
  ]);
  
  return ContentService.createTextOutput(
    JSON.stringify({status:"success"})
  ).setMimeType(ContentService.MimeType.JSON);
}
